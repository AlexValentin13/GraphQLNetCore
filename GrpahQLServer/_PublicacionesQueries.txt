# Obtiene Publicacion con Paginacion
query obtenerPublicaciones{
    publicaciones(
      after: "MQ==" # Implementación de paginaicón
      where: {
        id: {
          eq: 1
        }
      }
      order: {
        id:DESC
      }
    )
    {
      pageInfo
      {
        hasNextPage
        hasPreviousPage
        startCursor
        endCursor
      }
      nodes
      {
        id
        titulo
        contenido
        estado
      }
    }
}
